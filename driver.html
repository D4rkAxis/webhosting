FETCH DRIVER INFO (Plate & Phone) const driversSheet = ss.getSheetByName("DRIVERS"); if (driversSheet) { const driverData = driversSheet.getDataRange().getValues(); // Find driver by Name (Column 4, index 3) const driverRow = driverData.find(r => r[3] === selectedDriver); if (driverRow) { // Populate Card: Name(3), Plate(4), Phone(5), Joined(6) dash.getRange("H4").setValue(driverRow[3]).setFontWeight("bold"); dash.getRange("H5").setValue(driverRow[4]); dash.getRange("H6").setValue(driverRow[5]); dash.getRange("H7").setValue(driverRow[6]).setNumberFormat("yyyy-mm-dd"); } } // 2. FETCH TRIP ACTIVITY const trips = tripsSheet.getDataRange().getValues().slice(1).reverse(); // Skip header and show newest first const filtered = trips.filter(r => { if (String(r[2]) !== String(selectedDriver)) return false; // Check Name if (selectedDate) { // Robust Date Comparison (Handle both Date objects and Strings) const tripDateStr = String(r[3]).split(' ')[0]; // DB Date is "dd/MM/yyyy" let selDateStr = ""; if (selectedDate instanceof Date) { selDateStr = Utilities.formatDate(selectedDate, "GMT+1", "dd/MM/yyyy"); } else { selDateStr = String(selectedDate).split(' ')[0]; } if (tripDateStr !== selDateStr) return false; } return true; }); if (filtered.length === 0) return; // Map to Dashboard Columns const output = filtered.map(r => { const datePart = r[3].split(' ')[0]; const timeStart = r[3].split(' ')[1]; const timeStop = r[5] ? r[5].split(' ')[1] : "ONGOING"; // Generate Route Link (Origin -> Dest) let mapLink = ""; if (r[4] && r[6]) { mapLink = `=HYPERLINK("https://www.google.com/maps/dir/?api=1&origin=${r[4]}&destination=${r[6]}", "VIEW ROUTE")`; } else if (r[4]) { mapLink = `=HYPERLINK("https://maps.google.com/?q=${r[4]}", "VIEW START")`; } else { mapLink = "-"; } return [r[0], r[2], datePart, timeStart, timeStop, r[8], r[7], r[9], mapLink]; }); // Write Data dash.getRange(15, 1, output.length, 9).setValues(output); // Apply Professional Formatting to Data dash.getRange(15, 1, output.length, 9) .setHorizontalAlignment("center") .setVerticalAlignment("middle") .setWrapStrategy(SpreadsheetApp.WrapStrategy.WRAP) // Ensure text wraps instead of being cut off .setBorder(true, true, true, true, true, true, "#cbd5e1", SpreadsheetApp.BorderStyle.SOLID); } // ====================== API HANDLERS ====================== function doGet(e) { setup(false); // Ensure sheets exist const action = e.parameter.action; // Allow simple ping or basic get, but prefer POST for data to avoid caching issues try { return jsonOut({error: "Invalid Action"}); } catch (err) { return jsonOut({error: err.message}); } } function doPost(e) { setup(false); // Ensure sheets exist try { const data = JSON.parse(e.postData.contents); if (data.action === "login") { return loginDriver(data.username, data.password); } if (data.action === "getHistory") { return getTripHistory(data.driverId); } if (data.action === "register") { return registerDriver(data); } if (data.action === "startTrip") { return startTrip(data); } if (data.action === "stopTrip") { return stopTrip(data); } return jsonOut({error: "Invalid Action"}); } catch (err) { return jsonOut({error: err.message}); } } // ====================== CORE FUNCTIONS ====================== function registerDriver(data) { const ss = getDb(); const sheet = ss.getSheetByName("DRIVERS"); if (!sheet) throw new Error("DRIVERS sheet not found. Please run Setup."); const rows = sheet.getDataRange().getValues(); // Check if username exists const exists = rows.some(r => String(r[1]).toLowerCase() === String(data.username).toLowerCase()); if (exists) throw new Error("Username already taken"); const newId = "DRV-" + new Date().getTime().toString().slice(-6); sheet.appendRow([newId, data.username, data.password, data.fullName, data.carPlate, data.phone, new Date()]); updateDriverDropdown(ss); // Update dashboard dropdown return jsonOut({success: true, message: "Driver Registered!"}); } function loginDriver(username, password) { const ss = getDb(); const sheet = ss.getSheetByName("DRIVERS"); const rows = sheet.getDataRange().getValues(); const cleanUser = String(username).trim().toLowerCase(); const cleanPass = String(password).trim(); for (let i = 1; i < rows.length; i++) { if (String(rows[i][1]).trim().toLowerCase() === cleanUser && String(rows[i][2]).trim() === cleanPass) { return jsonOut({ success: true, driverId: rows[i][0], fullName: rows[i][3], carPlate: rows[i][4] }); } } throw new Error("Invalid Username or Password"); } function startTrip(data) { const ss = getDb(); const sheet = ss.getSheetByName("TRIPS"); const tripId = "TRIP-" + new Date().getTime().toString().slice(-8); const now = Utilities.formatDate(new Date(), "GMT+1", "dd/MM/yyyy HH:mm:ss"); // Columns: TripID, DriverID, DriverName, StartTime, StartGPS, StopTime, StopGPS, Duration, Distance, Status sheet.appendRow([tripId, data.driverId, data.driverName, now, data.gps, "", "", "", "", "ONGOING"]); return jsonOut({success: true, tripId: tripId, startTime: now}); } function stopTrip(data) { const ss = getDb(); const sheet = ss.getSheetByName("TRIPS"); const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (String(rows[i][0]) === String(data.tripId)) { const rowIdx = i + 1; const now = Utilities.formatDate(new Date(), "GMT+1", "dd/MM/yyyy HH:mm:ss"); // Calculate Distance (Haversine) const startGPS = rows[i][4].split(","); const stopGPS = data.gps.split(","); const dist = calculateDistance(startGPS[0], startGPS[1], stopGPS[0], stopGPS[1]); // Calculate Duration const t1 = parseDate(rows[i][3]); const t2 = new Date(); const diffMins = Math.round((t2 - t1) / 60000); sheet.getRange(rowIdx, 6).setValue(now); // StopTime sheet.getRange(rowIdx, 7).setValue(data.gps); // StopGPS sheet.getRange(rowIdx, 8).setValue(diffMins); // Duration (Mins) sheet.getRange(rowIdx, 9).setValue(dist.toFixed(2)); // Distance KM sheet.getRange(rowIdx, 10).setValue("COMPLETED"); // Status return jsonOut({success: true, distance: dist.toFixed(2), duration: diffMins, stopTime: now}); } } throw new Error("Trip ID not found"); } function getTripHistory(driverId) { const ss = getDb(); const sheet = ss.getSheetByName("TRIPS"); const rows = sheet.getDataRange().getValues(); // Filter by DriverID (Index 1) const myTrips = rows.slice(1).filter(r => String(r[1]) === String(driverId)).reverse(); return jsonOut(myTrips); } // ====================== HELPERS ====================== function jsonOut(data) { return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON); } function calculateDistance(lat1, lon1, lat2, lon2) { const R = 6371; // Radius of the earth in km const dLat = deg2rad(lat2 - lat1); const dLon = deg2rad(lon2 - lon1); const a = Math.sin(dLat/2) * Math.sin(dLat/2) + Math.cos(deg2rad(lat1)) * Math.cos(deg2rad(lat2)) * Math.sin(dLon/2) * Math.sin(dLon/2); const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a)); return R * c; // Distance in km } function parseDate(str) { const [date, time] = str.split(' '); const [d, m, y] = date.split('/'); const [h, min, s] = time.split(':'); return new Date(y, m - 1, d, h, min, s); } function deg2rad(deg) { return deg * (Math.PI/180); } function showAddDriverUi() { const ui = SpreadsheetApp.getUi(); const u = ui.prompt("Username").getResponseText();
